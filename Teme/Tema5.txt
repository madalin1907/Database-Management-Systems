DECLARE
    TYPE VECTOR IS VARRAY(5) OF EMP_FMA.EMPLOYEE_ID%TYPE;
    NR_ORDINE VECTOR;
    SALARIU_VECHI INTEGER;
    SALARIU_NOU INTEGER;
BEGIN
    SELECT EMPLOYEE_ID BULK COLLECT INTO NR_ORDINE
    FROM (SELECT EMPLOYEE_ID FROM EMP_FMA
          WHERE COMMISSION_PCT IS NULL
          ORDER BY SALARY ASC)
    WHERE ROWNUM <= 5;

    FOR I IN 1..5 LOOP
        SELECT SALARY INTO SALARIU_VECHI FROM EMP_FMA WHERE EMPLOYEE_ID = NR_ORDINE(I);
        UPDATE EMP_FMA SET SALARY = SALARY * 1.05 WHERE EMPLOYEE_ID = NR_ORDINE(I);
        SELECT SALARY INTO SALARIU_NOU FROM EMP_FMA WHERE EMPLOYEE_ID = NR_ORDINE(I);

        DBMS_OUTPUT.PUT_LINE('Salariu vechi: ' || SALARIU_VECHI || ', Salariu nou: ' || SALARIU_NOU);
    END LOOP;

    ROLLBACK;
END;
